: listATCGames
  getFilesInDirectory("games") -> Array gameFilenames
  
  gameFilenames.headIter -> Iter iter
  begin
  while( iter.next )
    -> String gameFilename
    "   " %s gameFilename.get %s %nl
    oclear gameFilename
  repeat
  oclear iter
  
  oclear gameFilenames
;

: atc
  openTrace
  getConsoleColor -> int color
  new atcGame -> game
  game.init( blword )
  srand(xor(time))
  if(atcFileReader:loadAtcFile( game.name.get game))
    clearConsole
    //t{ game.show "\n\n\n=====================================\n\n\n" %s }t
    game.play
    //t{ game.show }t
    game.reset
  else
    "Available games:" %s %nl
    listATCGames
  endif
  //t{ game.show }t
  oclear game
  setConsoleColor( color )
  closeTrace
;

: listGames
  int dirHandle
  ptrTo dirent pEntry
  int pName   //ptrTo byte pName
  
  opendir("games") -> dirHandle
  if( dirHandle )
    begin
      readdir(dirHandle) -> pEntry
    while( pEntry )
      pEntry.d_name(0 ref) -> pName // ref pName !
      //pEntry %x %bl pName %x %bl pName %s %nl
      if(pName c@ `.` <>)
        "   " %s pName %s %nl
      endif
    repeat
    closedir(dirHandle) drop
  endif
;
