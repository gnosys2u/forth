struct: sRGBA
  byte r
  byte g
  byte b
  byte alpha
  union
  int RGBA
;struct

class: rgba
  extends sRGBA
  
  method: set
	  -> alpha -> b -> g -> r
  ;method
  
  method: show
	  r %d %bl g %d %bl b %d %bl alpha %d %bl %nl
  ;method
  
  method: add
    // TOS is object to add
    -> rgba t
    t.r ->+ r
    t.g ->+ g
    t.b ->+ b
    t.alpha ->+ alpha
  ;method
  
;class
